	id = ww1_georgia.1
	hidden = yes

		tag = GEO

			has_cosmetic_tag = GEO_TSCN_commissariat

		}
		ARM = { exists = yes }
		AZR = { exists = yes }
	}
	immediate = {
        every_state = {
			limit = {
				OR = {
					is_core_of = ARM
					is_core_of = AZR
				}
				NOT = {
					state = 816
					state = 748
					state = 231
					state = 233
				}
			}
			ROOT = { remove_state_core = PREV }
		}
		drop_cosmetic_tag = yes
		set_party_name = {
			ideology = neutrality 
			long_name = GEO_neutrality_party 
			name = GEO_neutrality_party 
		}
		set_research_slots = 2
		set_stability = 0.56
		GEO_noe_ramishvili = {
			promote_character = socialism
			promote_character = authoritarian_democracy
		}
	}
}

# Transcaucasian Socialist Federative Soviet Republic
country_event = {
	id = ww1_georgia.9
	title = ww1_georgia.9.t
	desc = ww1_georgia.9.d
	picture = GFX_report_event_generic_communist_congress
	is_triggered_only = yes
	option = {
		name = ww1_georgia.9.a
		GEO = {
			ROOT = {
				every_owned_state = {
					limit = {
						OR = {
							is_core_of = GEO
							is_core_of = ARM
							is_core_of = AZR
						}
					}
					transfer_state_to = PREV.PREV
				}
			}
			every_state = {
				limit = {
					OR = {
						is_core_of = ARM
						is_core_of = AZR
					}
				}
				add_core_of = PREV
			}
			every_other_country = {
				limit = {
					is_subject_of = ROOT
					OR = {
						original_tag = ARM
						original_tag = AZR
					}
				}
				PREV = {
					annex_country = {
						target = PREV
						transfer_troops = yes
					}
				}
			}
			if = {
				limit = {
					NOT = {
						is_subject_of = ROOT
					}
				}
				ROOT = { puppet = PREV }
			}
			if = {
				limit = {
					has_government = communism
				}
				set_cosmetic_tag = GEO_TSCN_communism
				promote_character = GEO_sergo_ordzhonikidze
				else = {
					set_cosmetic_tag = GEO_TSCN
				}
			}		
		}
	}
}
