#Austrian answer to this all
country_event = {
	id = ww1_montenegro.9
	title = ww1_montenegro.9.t
	desc = ww1_montenegro.9.d
	is_triggered_only = yes
	fire_only_once = yes
	option = {
		#Jej
		name = ww1_montenegro.9.a
		MNT = {
			set_cosmetic_tag = MNT_SER
			every_country = {
				limit = {
					has_war_with = MNT
					NOT = {
						has_war_with = AUH
					}
				}
				white_peace = PREV
			}
			set_country_flag = MNT_serbia
			add_opinion_modifier = {
				target = SER
				modifier = faction_traitor
			}
			add_country_leader_trait = unpopular_monarch
		}
		every_state = {
			limit = {
				is_core_of = SER
			}
			add_core_of = MNT
		}
		every_state = {
			limit = {
				is_owned_by = SER
				is_controlled_by = AUH
			}
			MNT = {
				set_state_controller = PREV
			}
		}
		every_state = {
			limit = {
				is_owned_by = MNT
			}
			add_core_of = SER
		}
		if = {
			limit = {
				has_dlc = "Together for Victory"
			}
			set_autonomy = {
				target = MNT
				autonomy_state = autonomy_dominion
				freedom_level = 0.4
			}
			else = {
				GER = {
					add_to_faction = MNT
				}
			}
		}
		SER = {
			country_event = ww1_montenegro.10
			add_opinion_modifier = {
				target = MNT
				modifier = faction_traitor
			}
		}
		ai_chance = {
			factor = 40
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
	}
	option = {
		#he is still enemy tho
		name = ww1_montenegro.9.b
		ai_chance = {
			factor = 80
		}
	}
}
