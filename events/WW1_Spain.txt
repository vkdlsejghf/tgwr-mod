add_namespace = ww1_spain

#SPANISH REPUBLIC
country_event = {
	id = ww1_spain.1
	title = ww1_spain.1.t
	desc = ww1_spain.1.d
	picture = GFX_report_event_german_reichstag_gathering	
	fire_only_once = yes
	mean_time_to_happen = {	days = 30 }
	trigger = {
		original_tag = SPR
		OR = {
			has_government = neutrality
			AND = {
				has_government = fascism
				has_cosmetic_tag = SPR_neutrality
			}
		}
		OR = {
			has_country_flag = great_war_loser
			has_stability < 0.1
			democratic > 0.5
			AND = {
				date > 1931.4.1
				NOT = {
					has_country_leader = {
						character = SPR_miguel_primo_de_rivera
						ruling_only = yes
					}
				}
			}
		}
	}	
	option = {
		name = ww1_spain.1.a
		set_global_flag = spanish_republic
		set_cosmetic_tag = SPR_REP
		set_politics = {
			ruling_party = democratic 
			last_election = "1918.11.11"
			election_frequency = 60 
			elections_allowed = yes
		}
		hidden_effect = {
			set_popularities = {
				democratic = 43
				neutrality = 39
				fascism = 10
				communism = 8
			}
			if = {
				limit = {
					has_character = SPR_miguel_primo_de_rivera
				}
				retire_character = SPR_miguel_primo_de_rivera
				set_party_name = {
					ideology = fascism
					long_name = SPR_fascism_party
					name = SPR_fascism_party
				}
			}
		}
		ai_chance = { factor = 10 }
	}
}

# 1923 Spanish Coup
country_event = {
	id = ww1_spain.2
	title = ww1_spain.2.t
	desc = ww1_spain.2.d
	picture = GFX_report_event_spain_civil_war_soldiers
	fire_only_once = yes
	mean_time_to_happen = {	days = 13 }
	trigger = {
		tag = SPR
		has_character = SPR_miguel_primo_de_rivera
		has_country_flag = disaster_of_annual
		is_subject = no
		has_government = neutrality
		date > 1923.9.1
	}
	option = {
		name = ww1_spain.2.a
		hidden_effect = {
			set_country_flag = spanish_junta
			add_popularity = {
				ideology = fascism
				popularity = 0.3
			}
			set_cosmetic_tag = SPR_neutrality
		}
		SPR_miguel_primo_de_rivera = {
			add_country_leader_role = {
				promote_leader = yes
				country_leader = {
					ideology = stratocracy
				}
			}
		}
		add_ideas = SPR_alfonso_xiii
		set_party_name = {
			ideology = fascism
			long_name = SPR_military_directory
			name = SPR_military_directory
		}
		set_politics = {
			ruling_party = fascism
			elections_allowed = no
		}
		add_stability = -0.10
	}
}
