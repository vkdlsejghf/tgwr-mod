#Backstab Germany Austrian collapse
country_event = {
	id = ww1_austria.8
	title = ww1_austria.8.t
	desc = ww1_austria.8.d
	picture = GFX_report_event_ger_civil_war

	trigger = {
		original_tag = GER
		has_global_flag = interbellum
		has_country_flag = great_war_loser
		has_completed_focus = GER_backstab
		NOT = { country_exists = AUH }
		NOT = { country_exists = AUS }
		any_owned_state = {
			is_core_of = AUS
		}
		HUN = {
			NOT = {
				has_country_flag = hungarian_revanchism
			}
		}
	}
	fire_only_once = yes
	mean_time_to_happen = { days = 43 }

	immediate = {
		hidden_effect = {
			AUS = {
				add_state_core = 4
				add_state_core = 152
				add_state_core = 153
				add_state_core = 39
				add_state_core = 147
			}
			every_owned_state = {
				limit = {
					OR = {
						is_core_of = AUS
						is_core_of = SLV
						state = 244
					}
				}
				AUS = {
					transfer_state = PREV
				}
				remove_core_of = ROOT
			}
			every_state = {
				limit = {
					OR = {
						is_core_of = AUH
						is_claimed_by = AUH
					}
				}
				remove_core_of = AUH
				remove_claim_by = AUH
			}
			if = {
				limit = {
					has_country_flag = ger_kingdom_of_austria
				}
				AUS = {
					set_politics = {
						ruling_party = neutrality
						elections_allowed = no
					}
					if = {
						limit = {
							has_global_flag = kis_franzferdinand_murdered
						}
						AUS_karl_i = {
							if = {
								limit = { has_ideology = autocracy }
								promote_character = autocracy
							}
							else = {
								add_country_leader_role = {
									country_leader = {
										ideology = autocracy
									}
									promote_leader = yes
								}
							}
						}
						else = {
							AUS_franz_ferdinand_i = {
								if = {
									limit = { has_ideology = autocracy }
									promote_character = autocracy
								}
								else = {
									add_country_leader_role = {
										country_leader = {
											ideology = autocracy
										}
										promote_leader = yes
									}
								}
							}
						}
					}
				}
				else = {
					AUS = {
						promote_character = AUS_karl_seitz
						set_politics = {
							ruling_party = democratic
							elections_allowed = yes
						}
					}
				}
			}				
		}
	}
	#ok
	option = {
		name = ww1_austria.8.a
		ai_chance = { factor = 100 }
		news_event = { id = ww1_austria.100 }
	}
	#play as austria
	option = {
		name = ww1_austria.8.b
		ai_chance = { factor = 0 }
		news_event = { id = ww1_austria.100 }
		AUS = { change_tag_from = ROOT }
	}
}
