# Kingdom of Iraq
country_event = {
	id = ww1_iraq.1
	title = ww1_iraq.1.t
	desc = ww1_iraq.1.d
	picture = GFX_report_event_tur_turkish_soldiers
	fire_only_once = yes
	mean_time_to_happen = {
		days = 23
	}
	trigger = {
		tag = ENG
		owns_state = 291
		NOT = {
			country_exists = SYR
			country_exists = IRQ
		}
		date > 1921.8.1
		has_global_flag = interwar_iraqi_revolt
	}
	option = {
		name = ww1_iraq.1.a
		set_global_flag = kingdom_of_iraq
		IRQ = {
			hidden_effect = {
				SYR = {
					SYR_faisal_i = {
						set_nationality = IRQ
					}
				}
			}
			IRQ_faisal_i = {
				add_country_leader_role = {
					promote_leader = yes
					country_leader = {
						ideology = autocracy
					}
				}
			}
		}
	}
	option = {
		name = ww1_iraq.1.b
		every_owned_state = {
			limit = {
				is_core_of = IRQ
			}
			transfer_state_to = IRQ
		}
		declare_war_on = {
			target = IRQ
			type = annex_everything
		}
		ai_chance = {
			factor = 0
		}
	}
}

# Iraqi Independence
country_event = {
	id = ww1_iraq.2
	title = ww1_iraq.2.t
	desc = ww1_iraq.2.d
	picture = GFX_report_event_tur_turkish_soldiers
	fire_only_once = yes
	mean_time_to_happen = {
		days = 3
	}
	trigger = {
		tag = ENG
		owns_state = 291
		NOT = {
			country_exists = IRQ
		}
		has_global_flag = kingdom_of_iraq
		date > 1932.8.1
	}
	option = {
		name = ww1_iraq.2.a
		release = IRQ
		every_state = {
			limit = {
				is_core_of = IRQ
				is_claimed_by = ROOT
			}
			remove_claim_by = ROOT
		}
		if = {
			limit = {
				has_subject_autonomy_feature = yes
			}
			set_autonomy = {
				target = IRQ
				autonomy_state = autonomy_tgwr_informal_protectorate
				freedom_level = 0.8
			}
			IRQ = { set_country_flag = protectorate }
			else = {
				puppet = IRQ
				IRQ = { add_ideas = protectorate }
			}
		}
		hidden_effect = {
			IRQ = {
				set_politics = {
					ruling_party = neutrality
					elections_allowed = no
				}
				set_popularities = {
					fascism = 20
					neutrality = 80
				}
			}
		}
	}
	option = {
		name = ww1_iraq.1.b
		every_owned_state = {
			limit = {
				is_core_of = IRQ
			}
			transfer_state_to = IRQ
		}
		declare_war_on = {
			target = IRQ
			type = annex_everything
		}
		ai_chance = {
			factor = 0
		}
	}
}

			original_tag = SAU
		}
	}
	mean_time_to_happen = { days = 130 }
	option = {
		name = ww1_iraq.3.a
		every_owned_state = {
			limit = {
				OR = {
					is_core_of = IRQ
					is_core_of = KUR
				}
				NOT = { is_core_of = ROOT }
			}
			add_compliance = -70
			add_resistance = 50
		}
		news_event = interwar_news.14
	}
}
