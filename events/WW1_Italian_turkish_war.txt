		}

			add_stability = 0.05

			add_to_variable = {
				opda_lira = -2.950
			}
			add_to_variable = {
				opda_consumer_goods = -0.1
			}
			add_to_variable = {
				opda_resources_to_market = -0.05
			}
		}
		ITA = {
			add_stability = -0.1
			add_war_support = -0.1
			add_timed_idea = {
				idea = ITA_war_reparations
				days = 365
			}
		}
		every_country = {
			limit = {
				OR = {
					is_subject_of = TUR
					is_in_faction_with = TUR
				}
			}
			white_peace = ITA
		}
		every_country = {
			limit = {
				OR = {
					is_subject_of = ITA
					is_in_faction_with = ITA
				}
			}
			white_peace = TUR
		}
		hidden_effect = {
			ITA = {
				# Where is it?
				# country_event = {
				# 	days = 7
				# 	id = ww1_italy.10
				# }
			}
			TUR = {
				if = {
					limit = {
						TUR = {
					        has_country_flag = TUR_libya_set_port_4047
						}
					}
			        661 = {
				         remove_building = {
					        type = naval_base
					        province = 4047
					        level = 1
				        }
			        }
				}
				if = {
					limit = {
						TUR = {
					        has_country_flag = TUR_libya_set_port_1198
						}
					}
			        663 = {
				        remove_building = {
					        type = naval_base
					        province = 1198
					        level = 1
				        }
			        }
				}
			    if = {
				    limit = {
					    has_country_flag = TUR_italo_turkish_war_arab_minorities
				    }
			        every_state = {
				        limit = {
					        is_core_of = LBA
					        has_state_flag = TUR_removed_libyan_minority
				        }
				        add_dynamic_modifier = { modifier = TUR_arab_minority }
				        set_state_flag = TUR_arab_state
			        }
			    }
			}
			TUR = {
			    delete_unit_template_and_units = { division_template = "Trablusgarp Milisi" }
			}
		}
		set_global_flag = italo_turkish_war_ends
		news_event = ww1_italianturkish.9
	}
}
